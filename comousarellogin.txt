// Login y API (front servido desde la API en http://localhost:3000)
//
// 1) Levantar la API: cd Proyecto/api && npm start
//    Abrí http://localhost:3000/ → redirige a login.html
//
// 2) En Proyecto/api/.env definí JWT_SECRET, JWT_REFRESH_SECRET y FRONT_ORIGIN=http://localhost:3000
//
// 3) Ejecutar migración (solo la primera vez):
//    psql -h localhost -U postgres -d fcad_cursos -f Proyecto/api/migrations/001_refresh_tokens.sql
//
// 4) Login (desde login.html o consola del navegador en http://localhost:3000):
const API_BASE = 'http://localhost:3000';
const nombreUsuario = 'lbianchi'; // usuario del dump (ver dump base programacion.sql)
const contrasenia = 'TU_CONTRASENIA'; // contraseña en texto plano (se hashea con SHA-256 en el servidor)

const loginRes = await fetch(API_BASE + '/api/v2/auth/login', {
  method: 'POST',
  credentials: 'include',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({ nombreUsuario, contrasenia }),
});
const { accessToken, user } = await loginRes.json();
sessionStorage.setItem('accessToken', accessToken);
// El refresh token queda en cookie httpOnly (refreshToken); no acceder desde JS.
//
// 5) Requests con access token:
fetch(API_BASE + '/api/v2/estudiantes', {
  credentials: 'include',
  headers: { Authorization: 'Bearer ' + sessionStorage.getItem('accessToken') },
});
//
// 6) Renovar sesión manualmente (cookie refresh requerida):
const refreshRes = await fetch(API_BASE + '/api/v2/auth/refresh', {
  method: 'POST',
  credentials: 'include',
});
const { accessToken: nuevoAccess } = await refreshRes.json();
sessionStorage.setItem('accessToken', nuevoAccess);
//
// 7) Usuario autenticado:
fetch(API_BASE + '/api/v2/auth/me', {
  credentials: 'include',
  headers: { Authorization: 'Bearer ' + sessionStorage.getItem('accessToken') },
});
//
// 8) Cerrar sesión:
await fetch(API_BASE + '/api/v2/auth/logout', {
  method: 'POST',
  credentials: 'include',
});
sessionStorage.removeItem('accessToken');
//
// Notas
// - Access token corto (~15m). Refresh token en cookie httpOnly (~7d) con rotación en servidor.
// - El front (api.js) renueva el access token automáticamente ante 401.
// - Cursos, estudiantes e inscripciones requieren Bearer salvo POST /api/v2/auth/login|refresh|logout.
// - JWT_SECRET y JWT_REFRESH_SECRET deben ser cadenas largas y distintas en producción.
// - Rate limit en login: 5 intentos por IP cada 15 minutos.
